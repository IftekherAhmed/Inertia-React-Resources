5. Adding Sorting

Backend Changes

Update controller for sorting:

public function index(Request $request)
{
    $perPage = $request->input('per_page', 10);
    $search = $request->input('search');
    $companyId = $request->input('company_id');
    $sortBy = $request->input('sort_by', 'id');
    $sortDirection = $request->input('sort_direction', 'desc');
    
    $query = ModelsClient::query()
        ->with('company');
    
    if ($search) {
        $query->where(function($q) use ($search) {
            $q->where('name', 'like', "%$search%")
              ->orWhere('email', 'like', "%$search%")
              ->orWhere('phone', 'like', "%$search%");
        });
    }
    
    if ($companyId && $companyId !== 'all') {
        $query->where('company_id', $companyId);
    }
    
    if ($sortBy === 'company') {
        $query->leftJoin('company', 'client.company_id', '=', 'company.id')
            ->orderBy('company.name', $sortDirection)
            ->select('client.*');
    } else {
        $query->orderBy($sortBy, $sortDirection);
    }
    
    $clients = $query->paginate($perPage);
    
    return Inertia::render('Client/Index', [
        'client_list' => $clients->items(),
        'company_list' => Company::orderBy('name', 'asc')->get(),
        'pagination' => [
            'current_page' => $clients->currentPage(),
            'last_page' => $clients->lastPage(),
            'per_page' => $clients->perPage(),
            'total' => $clients->total(),
        ],
        'filters' => [
            'search' => $search,
            'company_id' => $companyId,
            'sort_by' => $sortBy,
            'sort_direction' => $sortDirection,
        ],
    ]);
}

// In Client controller
public function bulkDelete(Request $request)
{
    $request->validate([
        'ids' => 'required|array',
        'ids.*' => 'exists:client,id',
    ]);
    
    try {
        ModelsClient::whereIn('id', $request->ids)->delete();
        return back()->with('success', 'Selected clients deleted successfully!');
    } catch (Exception $e) {
        return back()->with('error', 'Error deleting clients: ' . $e->getMessage());
    }
}
